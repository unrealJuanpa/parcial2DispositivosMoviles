import 'package:flutter/material.dart';

void main() {
  runApp(MaterialApp(
    debugShowCheckedModeBanner: false,
    home: MyApp(),
  ));
}

class Registro {
  String titulo;
  int valor;
  DateTime fecha;

  Registro({required this.titulo, required this.valor, required this.fecha});
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(
          title: Text('CRUD'),
        ),
        body: Center(
            child: Column(
          children: <Widget>[
            Center(
              child: Text("hola"),
            ),
            Center(
              child: ElevatedButton(
                  onPressed: () {
                    _mostrarModal(context);
                  },
                  child: Text("Crear Registro")),
            )
          ],
        )),
      ),
    );
  }
}

void _mostrarModal(BuildContext context) {
  showDialog(
    context: context,
    builder: (BuildContext context) {
      return AlertDialog(
        title: Text('Crear Registro'),
        content: Text("Ingrese los datos para el nuevo registro"),
        actions: <Widget>[
          TextButton(
            onPressed: () {
              Navigator.of(context).pop();
            },
            child: Text('Guardar'),
          ),
          TextButton(
            onPressed: () {
              Navigator.of(context).pop();
            },
            child: Text('Cerrar'),
          ),
        ],
      );
    },
  );
}
